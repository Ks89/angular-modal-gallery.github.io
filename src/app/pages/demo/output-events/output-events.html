Code

                     (hasData)="onImageLoaded($event)"
                     (close)="onCloseImageModal($event)"
                     (show)="onVisibleIndex($event)"
                     (firstImage)="onIsFirstImage($event)"
                     (lastImage)="onIsLastImage($event)">
      </modal-gallery>
